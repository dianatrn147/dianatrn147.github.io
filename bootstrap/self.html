Self Portrait

This is a self portrait I made through p5.js

function setup() {
createCanvas(600,600);
background(146,216,233);
strokeWeight(10);
}

function draw() {
 //hair
fill(0);
arc(300, 400, 500, 750, PI, TWO_PI);
  
 //Head
fill(254,219,171);
ellipse(300, 250, 400, 400);

//glasses
fill(255,255,255);
strokeJoin(ROUND);
rect(190,200,80,80);
fill(255,255,255);
rect(320,200,80,80);
line(320,250,270,250);


//eyes
fill(0);
ellipse(230,250,20,20);
fill(0);
ellipse(360,250,20,20);

//body
square(150, 470, 300, 50, 50);

//mouth 
arc(300,350,70,50,0,3.14);


}
